# 星形 Star

Star 可绘制星型或正多边形

不设置 lineWidth 或 color 矩形不可见

# 属性

属性名 属性类型 默认值 说明
innerRadius number 0.5 设置内外半径比值
triangleNum number 5 设置角数
lineColor number 0xffffff 设置边框颜色
lineWidth number 0 设置边框宽度
lineType "full" , "dash" "full" 虚线or实线
color number 设置填充色
fillAlpha number 1 设置填充透明度
anchorX 0-1 设置内部X坐标
anchorY 0-1 设置内部Y坐标

# 定义

const star: gui.Star = {
    name: "star",
    type: guiType.Star,
    color:0xffffff,
    lineColor: 0xff00cc,
    lineWidth: 1,
    radius:10
};

# 使用

{
id: "star",
libId: ComponentId.star,
style: {
    width: 100,
    height: 100
}
}
