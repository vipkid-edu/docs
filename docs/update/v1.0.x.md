```
    if(msg.code === 'native') {
      if(msg.data.type === 'position' && msg.target) {
        // 视频位置
        const rect = msg.target.toDomRectangle();
        rect.x;
        rect.y;
        rect.width;
        rect.height;
      }
    }
  }
```
